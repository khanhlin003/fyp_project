\section{UC-01: Register Account}
\label{uc:register}

\begin{longtable}{|>{\raggedright\arraybackslash\bfseries}p{3.5cm}|p{11cm}|}
\hline
\rowcolor{gray!15} Use Case ID & UC-01 \\ \hline
Use Case Name & Register Account \\ \hline
Primary Actor & Retail Investor \\ \hline
Description & Enables a new user to create an account on the platform by providing valid registration credentials. \\ \hline
Preconditions &
\begin{itemize}[leftmargin=*, nosep]
    \item User has access to the platform via a supported web browser.
    \item User does not already have an account associated with the provided email address.
\end{itemize} \\ \hline
Main Flow &
\begin{enumerate}[leftmargin=*, nosep]
    \item User navigates to the registration page.
    \item User enters a valid email address.
    \item User enters a password with a minimum length of six characters.
    \item User confirms the password.
    \item User submits the registration form.
    \item System validates the input fields.
    \item System creates a new user account.
    \item System authenticates the user and redirects them to the risk questionnaire.
\end{enumerate} \\ \hline
Alternative Flows &
\textbf{AF-1: Email Already Registered} \newline
At Step 6, if the email already exists, the system displays an error message. \newline
\textbf{AF-2: Invalid Input} \newline
If validation fails, the system highlights invalid fields and prompts correction. \\ \hline
Postconditions &
\begin{itemize}[leftmargin=*, nosep]
    \item User account created and authenticated via JWT.
\end{itemize} \\ \hline
Business Rules &
\begin{itemize}[leftmargin=*, nosep]
    \item Password must be \textbf{\ge} 6 characters and securely hashed.
\end{itemize} \\ \hline
\end{longtable}

\hline
\rowcolor{gray!15} Use Case ID & UC-11 \\ \hline
Use Case Name & Manage Wallets \\ \hline
Primary Actor & Retail Investor \\ \hline
Description & User creates, updates, and deactivates goal-based wallets for organizing investments. \\ \hline
Preconditions &
\begin{itemize}[leftmargin=*, nosep]
    \item User is authenticated.
\end{itemize} \\ \hline
Main Flow &
\begin{enumerate}[leftmargin=*, nosep]
    \item User opens the wallet management page.
    \item User creates a new wallet by entering wallet name and purpose.
    \item System creates the wallet and initializes default wallet profile settings.
    \item User optionally updates wallet profile fields (risk profile, horizon, objective, and liquidity settings).
    \item User may deactivate an existing wallet.
\end{enumerate} \\ \hline
Alternative Flows &
\textbf{AF-1: Invalid Wallet Input} \newline
System prompts user to correct invalid values (for example name too short or invalid profile ranges). \\ \hline
Postconditions &
\begin{itemize}[leftmargin=*, nosep]
    \item Wallet records and wallet profile settings are updated for the authenticated user.
\end{itemize} \\ \hline
Business Rules &
\begin{itemize}[leftmargin=*, nosep]
    \item Wallet operations are user-scoped; users can only manage their own wallets.
    \item Deactivation marks wallet as inactive rather than deleting historical data.
\end{itemize} \\ \hline
\end{longtable}

\rowcolor{gray!15} Use Case ID & UC-12 \\ \hline
Use Case Name & Assign Holdings to Wallet \\ \hline
Primary Actor & Retail Investor \\ \hline
Description & User assigns a portfolio holding to a selected wallet for goal-based grouping and analytics. \\ \hline
Preconditions &
\begin{itemize}[leftmargin=*, nosep]
    \item User is authenticated.
    \item User has at least one existing wallet.
    \item User has at least one portfolio holding.
\end{itemize} \\ \hline
Main Flow &
\begin{enumerate}[leftmargin=*, nosep]
    \item User selects an existing holding from portfolio or wallet interface.
    \item User chooses a target wallet.
    \item System updates wallet assignment for the holding.
    \item System refreshes wallet holdings and portfolio summaries.
\end{enumerate} \\ \hline
Alternative Flows &
\textbf{AF-1: Wallet Not Found or Inactive} \newline

    \item Holding is mapped to the selected wallet and reflected in wallet-level views.
\end{itemize} \\ \hline
Business Rules &
\begin{itemize}[leftmargin=*, nosep]
    \item A holding can be mapped to one active wallet assignment at a time.
    \item Assignment changes trigger recalculation of wallet-level current value and allocation outputs.
\end{itemize} \\ \hline
\end{longtable}

\hline
\rowcolor{gray!15} Use Case ID & UC-10 \\ \hline
Use Case Name & View News Sentiment \\ \hline
Primary Actor & Retail Investor \\ \hline
Description & User views ETF and portfolio-related news with sentiment labels and alerts. \\ \hline
Preconditions &
\begin{itemize}[leftmargin=*, nosep]
    \item News data source is available.
    \item User has selected a ticker or portfolio context.
\end{itemize} \\ \hline
Main Flow &
\begin{enumerate}[leftmargin=*, nosep]
    \item User opens a ticker detail page or portfolio dashboard.
    \item System retrieves recent related news articles.
    \item System displays sentiment labels and relevance details.
    \item User reviews alerts and article links.
\end{enumerate} \\ \hline
Alternative Flows &
\textbf{AF-1: API Failure} \newline

Postconditions &
\begin{itemize}[leftmargin=*, nosep]

\end{itemize} \\ \hline
Business Rules &
\begin{itemize}[leftmargin=*, nosep]
    \item News windows are query-driven (for example last N days), not fixed to a single 24-hour window.
    \item Sentiment scores normalized between -1 and +1.
\end{itemize} \\ \hline
\end{longtable}
